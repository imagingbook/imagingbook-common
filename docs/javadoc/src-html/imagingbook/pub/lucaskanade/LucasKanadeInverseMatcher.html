009package imagingbook.pub.lucaskanade;
010
011import ij.IJ;
012import ij.process.FloatProcessor;
013import imagingbook.lib.ij.IjUtils;
014import imagingbook.lib.math.Matrix;
015import imagingbook.pub.geometry.basic.Point;
016import imagingbook.pub.geometry.mappings.linear.ProjectiveMapping2D;
017
018
019/**
020 * Inverse Compositional Matcher, as described in Simon Baker and Iain Matthews, 
021 * "Lucas-Kanade 20 Years On: A Unifying Framework: Part 1", CMU-RI-TR-02-16 (2002)
022 *  @author Wilhelm Burger
023 *  @version 2014/02/08
024 */
025public class LucasKanadeInverseMatcher extends LucasKanadeMatcher {
026        
027        private int n;                                  // number of warp parameters
028        private float[][] Rx, Ry;               // gradient of reference image
029        private double[][] Hi;                  // inverse of cumulated Hessian matrix
030        private double[][][] S;                 // S[u][u][n] = the steepest descent image for dimension n at pos. u,v (same size as R)
031
032        private double qmag = Double.MAX_VALUE;         // magnitude of parameter difference vector
033        private double sqrError = Double.MAX_VALUE;     // squared sum of differences between I and R
034        
035        /**
036         * Creates a new matcher of type {@link LucasKanadeInverseMatcher}.
037         * @param I the search image (of type {@link FloatProcessor}).
038         * @param R the reference image (of type {@link FloatProcessor})
039         * @param params a parameter object of type  {@link LucasKanadeMatcher.Parameters}.
040         */
041        public LucasKanadeInverseMatcher(FloatProcessor I, FloatProcessor R, Parameters params) {
042                super(I, R, params);
043        }
044        
045        public LucasKanadeInverseMatcher(FloatProcessor I, FloatProcessor R) {
046                super(I, R, new Parameters());
047        }
048        
049        private void initializeMatch(ProjectiveMapping2D Tinit) {
050//              n = Tinit.getParameters().length;       // number of transformation parameters
051                n = getParameters(Tinit).length;        // number of transformation parameters
052                S = new double[wR][hR][];                       // S[u][v] holds a double vector of length n
053                Rx = gradientX(R).getFloatArray();      // gradient of R
054                Ry = gradientY(R).getFloatArray();
055                double[][] H = new double[n][n];        // cumulated Hessian matrix of size n x n (initially zero)
056                
057                ProjectiveMapping2D Tp = Tinit;
058                
059                for (int u = 0; u < wR; u++) {                  // for all coordinates (u,v) in R do
060                        for (int v = 0; v < hR; v++) {
061//                              double[] x = {u - xc, v - yc};  // position w.r.t. the center of R
062                                Point x = Point.create(u - xc, v - yc); // position w.r.t. the center of R
063                                double[] gradR = {Rx[u][v], Ry[u][v]};
064
065                                double[][] J = Tp.getJacobian(x);
066                                
067                                double[] sx = Matrix.multiply(gradR, J);                // column vector of length n (6)
068                                
069                                S[u][v] = sx;                                                                   // keep for use in main loop
070                                double[][] Hxy = Matrix.outerProduct(sx, sx);
071                                
072                                H = Matrix.add(H, Hxy);                                                 // cumulate the Hessian
073                        }
074                }
075                
076                Hi = Matrix.inverse(H);                                                                 // inverse of Hessian
077                if (Hi == null) {
078                        IJ.log("singular Hessian!");
079                        throw new RuntimeException("could not invert Hessian");
080                }
081                
082                iteration = 0;
083                
084                if (params.showSteepestDescentImages) 
085                        showSteepestDescentImages(S);
086                if (params.showHessians) {
087                        IjUtils.createImage("H", H).show();
088                        IjUtils.createImage("Hi", Matrix.inverse(H)).show();
089                }
090        }
091        
092        @Override
093        public ProjectiveMapping2D iterateOnce(ProjectiveMapping2D Tp) {
094                if (iteration < 0) {
095                        initializeMatch(Tp);
096                }
097                iteration = iteration + 1;
098                double[] dp = new double[n];    // n-dim vector \delta_p = 0
099                sqrError = 0;
100                
101                // for all positions (u,v) in R do
102                for (int u = 0; u < wR; u++) {
103                        for (int v = 0; v < hR; v++) {
104                                
105                                // get coordinate relative to center of R
106//                              double[] x = {u - xc, v - yc};
107                                Point x = Point.create(u - xc, v - yc);
108                                
109                                // warp I to I' (onto R)
110//                              double[] xT = Tp.applyTo(x);            // warp from x -> x'
111                                Point xT = Tp.applyTo(x);               // warp from x -> x'
112
113                                // calculate pixel difference d for pos. (u,v)
114                                double d = I.getInterpolatedValue(xT.getX(), xT.getY()) - R.getf(u, v);
115                                sqrError = sqrError + d * d;
116                                
117                                // multiply the pixel difference d with the corresponding steepest descent image sx
118                                // and sum into dp:
119                                double[] sx = S[u][v];
120                                dp = Matrix.add(dp, Matrix.multiply(d, sx));
121                        }
122                }
123                
124                // estimate the parameter difference vector qopt:
125                double[] qopt = Matrix.multiply(Hi, dp);
126                if (params.debug) IJ.log("qopt  = " + Matrix.toString(qopt));
127                
128                // measure the magnitude of the difference vector:
129                qmag = Matrix.normL2squared(qopt);
130                
131                // Calculate the warp parameters p', such that T_p'(x) = T_p (T^-1_q (x), for any point x.
132//              ProjectiveMapping2D Tqopt = ProjectiveMapping2D.fromParameters(qopt);
133                ProjectiveMapping2D Tqopt = toProjectiveMap(qopt);
134                ProjectiveMapping2D Tqopti = Tqopt.getInverse();
135                return Tqopti.concat(Tp);
136        }
137        
138        @Override
139        public boolean hasConverged() {
140                return (qmag < params.tolerance);
141        }
142        
143        @Override
144        public double getRmsError() {
145                return Math.sqrt(sqrError);
146        }
147        
148}
