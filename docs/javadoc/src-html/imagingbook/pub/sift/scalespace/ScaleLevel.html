010package imagingbook.pub.sift.scalespace;
011
012import ij.process.FloatProcessor;
013//import imagingbook.pub.sift.filters.GaussianFilter;
014import imagingbook.lib.filters.GaussianFilter;
015
016/**
017 * Represents a single scale level. Just a special kind of FloatProcessor
018 * with some extra fields and methods.
019 */
020public class ScaleLevel extends FloatProcessor {        // TODO: make IJ independent, use only float arrays
021        
022        private double absoluteScale;   // really needed?
023        
024        // ------------------------------
025        
026        public ScaleLevel(int width, int height, float[] data, double absoluteScale) {
027                super(width, height, data, null);       // constructor of FloatProcessor
028                this.absoluteScale = absoluteScale;
029        }
030        
031        public ScaleLevel(FloatProcessor fp, double absoluteScale) {
032//              this(fp.getWidth(), fp.getHeight(), ((float[]) fp.getPixels()).clone(), absoluteScale);
033                this(fp.getWidth(), fp.getHeight(), getValues(fp, true), absoluteScale);
034        }
035        
036        public ScaleLevel(ScaleLevel level) {
037//              this(level.getWidth(), level.getHeight(), ((float[]) level.getPixels()).clone(), level.absoluteScale);
038                this(level.getWidth(), level.getHeight(), getValues(level, false), level.absoluteScale);
039        }
040        
041        private static float[] getValues(FloatProcessor fp, boolean normalize) {
042                final double minVal = fp.getMin();
043                final double maxVal = fp.getMax();
044                final float offset = (float) -minVal;
045                final float scale = (float) (1.0 / (maxVal - minVal));
046                final float[] values = (float[]) fp.getPixelsCopy();
047                if (normalize) {
048                        for (int i = 0; i < values.length; i++) {
049                                values[i] = (values[i] + offset) * scale; 
050                        }
051                }
052                return values;
053        }
054        
055        // ------------------------------
056
057        public void filterGaussian(double sigma) {
058                GaussianFilter gf = new GaussianFilter(sigma);
059                gf.applyTo(this);
060        }
061        
062        public ScaleLevel duplicate() {
063                return new ScaleLevel(this);
064        }
065
066        public ScaleLevel decimate() {  // returns a 2:1 subsampled copy of this ScaleLevel
067                //IJ.log(" reducing ...");
068                int width1 = this.getWidth();
069                int height1 = this.getHeight();
070                int width2 = width1 / 2;
071                int height2 = height1 / 2;
072                
073                float[] pixels1 = (float[]) this.getPixels();
074                float[] pixels2 = new float[width2*height2];            
075                for (int v2 = 0 ; v2 < height2; v2++) {
076                        int v1 = 2 * v2;
077                        for (int u2 = 0 ; u2 < width2; u2++) {
078                                int u1 = 2 * u2;
079                                pixels2[v2 * width2 + u2] = pixels1[v1 * width1 + u1];
080                        }
081                }
082                return new ScaleLevel(width2, height2, pixels2, absoluteScale);
083        }
084        
085        public ScaleLevel subtract(FloatProcessor B) {
086                // A <-- A-B
087                ScaleLevel A = this.duplicate();
088                float[] pixelsA = (float[]) A.getPixels();
089                float[] pixelsB = (float[]) B.getPixels();
090                for (int i=0; i<pixelsA.length; i++) {
091                        pixelsA[i] = pixelsA[i] - pixelsB[i];
092                }
093                A.setAbsoluteScale(0);
094                return A;
095        }
096        
097        public void setAbsoluteScale(double sigma) {
098                this.absoluteScale = sigma;
099        }
100        
101        public double getAbsoluteScale() {
102                return absoluteScale;
103        }
104        
105        /*
106         * Collects and returns the 3x3 neighborhood values at this scale level 
107         * at center position u,v. The result is stored in the given 3x3 array nh.
108         */
109        public void get3x3Neighborhood(final int u, final int v, final float[][] nh) {
110                for (int i = 0, x = u - 1; i < 3; i++, x++) {
111                        for (int j = 0, y = v - 1; j < 3; j++, y++) {
112                                nh[i][j] = this.getf(x, y);
113                        }
114                }
115        }
116        
117        public void getGradientPolar(int u, int v, double[] grad) {
118                final double grad_x = this.getf(u+1, v) - this.getf(u-1, v);    // x-component of local gradient
119                final double grad_y = this.getf(u, v+1) - this.getf(u, v-1);    // y-component of local gradient
120                grad[0] = Math.sqrt(grad_x * grad_x + grad_y * grad_y);         // local gradient magnitude (E)
121                grad[1] = Math.atan2(grad_y, grad_x);                                           // local gradient direction (phi)
122        }
123}
