020/**
021 * Binary region labeler based on a combined region labeling
022 * and contour tracing algorithm:
023 * F. Chang, C. J. Chen, and C. J. Lu. A linear-time component labeling
024 * algorithm using contour tracing technique. Computer Vision, Graphics,
025 * and Image Processing: Image Understanding 93(2), 206-220 (2004).
026 * 
027 * @author WB
028 * @version 2020/04/01
029 */
030public class RegionContourLabeling extends RegionLabeling implements ContourTracer { 
031        
032        static private final int VISITED = -1;
033        
034        private List<Contour> outerContours;
035        private List<Contour> innerContours;
036        
037        /**
038         * Creates a new region labeling.
039         * 
040         * @param ip the binary input image with 0 values for background pixels and values &gt; 0
041         * for foreground pixels.
042         */
043        public RegionContourLabeling (ByteProcessor ip) {
044                super(ip);      // all work is done by the constructor of the superclass
045        }

066                return copyContours(innerContours, sort);

070        // non-public methods
071        
072        @Override
073        protected int[][] initialize() {
074                // Create a label array which is "padded" by 1 pixel, i.e., 
075                // 2 rows and 2 columns larger than the image:
076                int[][] lA = new int[width+2][height+2];        // label array, initialized to zero
077                outerContours = new ArrayList<Contour>();
078                innerContours = new ArrayList<Contour>();
079                return lA;
080        }
081        
082        @Override
083        void applyLabeling() {
084                resetLabel();
085                // scan top to bottom, left to right
086                for (int v = 0; v < height; v++) {
087                        int label = 0;  // reset label, scan through horiz. line:
088                        for (int u = 0; u < width; u++) {
089                                if (ip.getPixel(u, v) > 0) {    // unlabeled FOREGROUND pixel
090                                        if (label != 0) { // keep using the same label
091                                                setLabel(u, v, label);
092                                        }
093                                        else {  // label == zero
094                                                label = getLabel(u, v);
095                                                if (label == 0) {       // new (unlabeled) region is hit
096                                                        label = getNextLabel(); // assign a new region label
097                                                        Contour oc = traceContour(u, v, 0, label);
098                                                        outerContours.add(oc);
099                                                        setLabel(u, v, label);
100                                                }
101                                        }
102                                } 
103                                else {  // BACKGROUND pixel
104                                        if (label != 0) { // exiting a region
105                                                if (getLabel(u, v) == BACKGROUND) { // unlabeled - new inner contour
106                                                        Contour ic = traceContour(u - 1, v, 1, label);
107                                                        innerContours.add(ic);
108                                                }
109                                                label = 0;
110                                        }
111                                }
112                        }
113                }
114        }
115        
116        @Override
117        void collectRegions() {
118                super.collectRegions(); // collect region pixels and calculate statistics
119                attachOuterContours();  // attach each outer contours to the corresponding region
120                attachInnerContours();  // attach all inner contours to the corresponding regions
121        }
122        
123        // Trace one contour starting at (xS,yS) 
124        // in direction dS with label label
125        // trace one contour starting at (xS,yS) in direction dS        
126        private Contour traceContour(int xS, int yS, int dS, int label) {
127                Contour contr = new Contour(label);
128                int xT, yT; // T = successor of starting point (xS,yS)
129                int xP, yP; // P = previous contour point
130                int xC, yC; // C = current contour point
131//              Point pt = Point.create(xS, yS); 
132                int[] pt = {xS, yS};
133                int dNext = findNextPoint(pt, dS);
134//              contr.addPoint(pt); 
135                contr.addPoint(Point.create(pt[0], pt[1]));
136                xP = xS; yP = yS;
137                xC = xT = pt[0];
138                yC = yT = pt[1];
139                
140                boolean done = (xS == xT && yS == yT);  // true if isolated pixel
141                while (!done) {
142                        setLabel(xC, yC, label);
143//                      pt = Point.create(xC, yC);
144                        int[] pn = {xC, yC};
145                        int dSearch = (dNext + 6) % 8;
146                        dNext = findNextPoint(pn, dSearch);
147                        xP = xC;  yP = yC;      
148                        xC = pn[0]; 
149                        yC = pn[1]; 
150                        // are we back at the starting position?
151                        done = (xP==xS && yP==yS && xC==xT && yC==yT);
152                        if (!done) {
153                                contr.addPoint(Point.create(pn[0], pn[1]));
154                        }
155                }
156                return contr;
157        }

183        private int findNextPoint (int[] pos, int dir) {        // TODO: PROBLEM!!
184                // Starts at point pos in direction dir,
185                // returns the resulting tracing direction
186                // and modifies pt.
187                for (int i = 0; i < 7; i++) {
188                        int x = pos[0] + delta[dir][0];
189                        int y = pos[1] + delta[dir][1];
190                        if (ip.getPixel(x, y) == BACKGROUND) {
191                                setLabel(x, y, VISITED);        // mark surrounding background pixels
192                                dir = (dir + 1) % 8;
193                        } 
194                        else {  // found a non-background pixel (next pixel to follow)
195                                pos[0] = x; 
196                                pos[1] = y; 
197                                break;
198                        }
199                }
200                return dir;
201        }
202        
203        private void attachOuterContours() {
204                for (Contour c : outerContours) {
205                        int label = c.getLabel();
206                        BinaryRegion r = findRegion(label);
207                        if (r == null) {
208                                IJ.log("Error: Could not associate outer contour with label " + label);
209                        }
210                        else {
211                                r.setOuterContour(c);
212                        }
213                }
214        }
215        
216        private void attachInnerContours() {
217                for (BinaryRegion r : regions) {
218                        r.makeInnerContours();  // ensure that every region has a (empty) list of inner contours
219                }
220                for (Contour c : innerContours) {
221                        int label = c.getLabel();
222                        BinaryRegion r = findRegion(label);
223                        if (r == null) {
224                                IJ.log("Error: Could not associate inner contour with label " + label);
225                        }
226                        else {
227                                r.addInnerContour(c);
228                        }
229                }
230        }

232        // access methods to the label array (which is padded!)
233        @Override
234        public int getLabel(int u, int v) {     // (u,v) are image coordinates
235                if (u >= -1 && u <= width && v >= -1 && v <= height)
236                        return labelArray[u + 1][v + 1];        // label array is padded (offset = 1)
237                else
238                        return BACKGROUND;
239                //return labelArray[u + 1][v + 1];      // original version
240        }
241        
242        @Override
243        protected void setLabel(int u, int v, int label) { // (u,v) are image coordinates
244                if (u >= -1 && u <= width && v >= -1 && v <= height) {
245                        labelArray[u + 1][v + 1] = label;
246                }
247        }
248        
249        private List<Contour> copyContours(List<Contour> cntrs, boolean sort) {
250                if (cntrs == null)
251                        return Collections.emptyList(); 
252                else {
253                        List<Contour> cc = new ArrayList<Contour>(cntrs);
254                        if (sort) {
255                                Collections.sort(cc);
256                        }
257                        return cc;
258                }
259        }
