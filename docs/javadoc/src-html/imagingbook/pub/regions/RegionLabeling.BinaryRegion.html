032/**
033 * This class does the complete region labeling for a given image.
034 * It is abstract, because the implementation of some parts depends
035 * on the region labeling algorithm being used.
036 * 
037 * @version 2018/04/01
038 */
039public abstract class RegionLabeling {
040        
041        static final int BACKGROUND = 0;
042        static final int FOREGROUND = 1;
043        static final int START_LABEL = 2;
044
045        protected final ImageProcessor ip;
046        protected final int width;
047        protected final int height;
048        
049        protected final int[][] labelArray;
050        // label values in labelArray can be:
051        //  0 ... unlabeled
052        // -1 ... previously visited background pixel
053        // >0 ... valid label
054        
055        protected int currentLabel;
056        protected int maxLabel; // the maximum label in the labels array
057        protected List<BinaryRegion> regions;
058        
059        protected RegionLabeling(ByteProcessor ip) {
060                this.ip = ip;
061                width  = ip.getWidth();
062                height = ip.getHeight();
063                labelArray = initialize();
064                applyLabeling();
065                collectRegions();
066        }
067        
068        protected int[][]  initialize() {
069                int[][] lA = new int[width][height];    // label array
070                // set all pixels to either FOREGROUND or BACKGROUND (by thresholding)
071                for (int v = 0; v < height; v++) {
072                        for (int u = 0; u < width; u++) {
073                                lA[u][v] = (ip.getPixel(u, v) > 0) ? FOREGROUND : BACKGROUND;
074                        }
075                }
076                return lA;
077        }
078        
079        /**
080         * Get an unsorted list of all regions associated with this region labeling.
081         * @return the list of regions.
082         */
083        public List<BinaryRegion> getRegions() {
084                return getRegions(false);       // unsorted
085        }
086        
087        /**
088         * Get a possibly sorted list of all regions associated with this region labeling.
089         * @param sort set {@code true} to sort regions by decreasing size.
090         * @return the list of regions.
091         */
092        public List<BinaryRegion> getRegions(boolean sort) {
093                if (regions == null) 
094                        return Collections.emptyList();
095                else {
096                        List<BinaryRegion> rns = new ArrayList<BinaryRegion>(regions);
097                        if (sort) {
098                                Collections.sort(rns);
099                        }
100                        return rns;
101                }
102        }

104        // This method must be implemented by any real sub-class:
105        abstract void applyLabeling();
106        
107        // creates a container of BinaryRegion objects
108        // collects the region pixels from the label image
109        // and computes the statistics for each region
110        void collectRegions() {
111                BinaryRegion[] regionArray = new BinaryRegion[maxLabel + 1];
112                for (int i = START_LABEL; i <= maxLabel; i++) {
113                        regionArray[i] = new BinaryRegion(i);
114                }
115                for (int v = 0; v < height; v++) {
116                        for (int u = 0; u < width; u++) {
117                                int label = getLabel(u, v);
118                                if (label >= START_LABEL && label <= maxLabel && regionArray[label] != null) {
119                                        regionArray[label].addPixel(u, v);
120                                }
121                        }
122                }
123                // create a list of regions to return, collect nonempty regions
124                List<BinaryRegion> regionList = new LinkedList<BinaryRegion>();
125                for (BinaryRegion r: regionArray) {
126                        if (r != null && r.getSize() > 0) {
127                                r.update();     // compute the statistics for this region
128                                regionList.add(r);
129                        }
130                }
131                regions = regionList;
132        }
133        
134        /**
135         * Get the label number for the specified image coordinate.
136         * @param u the horizontal coordinate.
137         * @param v the vertical coordinate.
138         * @return the label number for the given position.
139         */
140        public int getLabel(int u, int v) {
141                if (u >= 0 && u < width && v >= 0 && v < height)
142                        return labelArray[u][v];
143                else
144                        return BACKGROUND;
145        }
146        
147        protected void setLabel(int u, int v, int label) {
148                if (u >= 0 && u < width && v >= 0 && v < height)
149                        labelArray[u][v] = label;
150        }
151        
152        protected void resetLabel() {
153                currentLabel = -1;
154                maxLabel = -1;
155        }
156        
157        int getNextLabel() {
158                if (currentLabel < 1)
159                        currentLabel = START_LABEL;
160                else
161                        currentLabel = currentLabel + 1;
162                maxLabel = currentLabel;
163                return currentLabel;
164        }
165        
166        int getMaxLabel() {
167                return maxLabel;
168        }

172        /**
173         * Utility method that creates an image of the internal
174         * label array.
175         * 
176         * @param color set {@code false} to get a 16-bit grayscale image,
177         * {@code true} for an RGB (24-bit) color image.
178         * @return an image of the label array.
179         */
180        public ImageProcessor makeLabelImage(boolean color) {
181                return (color) ?  makeLabelImageColor() : makeLabelImageGray();
182        }
183
184        private ColorProcessor makeLabelImageColor() {
185                int[] colorLUT = new int[maxLabel+1];
186                for (int i = START_LABEL; i <= maxLabel; i++) {
187                        colorLUT[i] = makeRandomColor();
188                }
189                ColorProcessor cp = new ColorProcessor(width, height);
190                for (int v = 0; v < height; v++) {
191                        for (int u = 0; u < width; u++) {
192                                int lb = getLabel(u, v);
193                                if (lb >= 0 && lb < colorLUT.length) {
194                                        cp.putPixel(u, v, colorLUT[lb]);
195                                }
196                        }
197                }
198                return cp;
199        }
200        
201        private ShortProcessor makeLabelImageGray() {
202                ShortProcessor sp = new ShortProcessor(width, height);
203                for (int v = 0; v < height; v++) {
204                        for (int u = 0; u < width; u++) {
205                                int lb = getLabel(u, v);
206                                sp.set(u, v, (lb >= 0) ? lb : 0);
207                        }
208                }
209                sp.resetMinAndMax();
210                return sp;
211        }

214        /**
215         * Find the region associated to the given label.
216         * @param label the label number.
217         * @return the region object associated with the given label
218         *              or {@code null} if it does not exist.
219         */
220        public BinaryRegion findRegion(int label) {
221                if (label <= 0 || regions == null) return null;
222                BinaryRegion regn = null;
223                for (BinaryRegion r : regions) {
224                        if (r.getLabel() == label) {
225                                regn = r;
226                                break;
227                        }
228                }
229                return regn;
230        }

232        /**
233         * Finds the {@link BinaryRegion} instance associated with
234         * the given image position.
235         * @param u the horizontal position.
236         * @param v the vertical position.
237         * @return The associated {@link BinaryRegion} object or null if
238         * this {@link RegionLabeling} has no region at the given position.
239         */
240        public BinaryRegion getRegion(int u, int v) {
241                int label = getLabel(u, v);
242                return findRegion(label);
243        }
244        
245        protected int makeRandomColor() {
246                double saturation = 0.2;
247                double brightness = 0.2;
248                float h = (float) Math.random();
249                float s = (float) (saturation * Math.random() + 1 - saturation);
250                float b = (float) (brightness * Math.random() + 1 - brightness);
251                return Color.HSBtoRGB(h, s, b);
252        }

280        private class RegionPixelIterator implements Iterator<Point> {
281                final int label;                                        // the corresponding region's label
282                final int uMin, uMax, vMin, vMax;       // coordinates of region's bounding box
283                int uCur, vCur;                                         // current pixel position
284                Point pNext;                                            // coordinates of the next region pixel
285                boolean first;                                          // control flag
286
287                RegionPixelIterator(BinaryRegion R) {
288                        label = R.getLabel();
289                        first = true;
290                        Rectangle bb = R.getBoundingBox();
291                        uMin = bb.x;
292                        uMax = bb.x + bb.width;
293                        vMin = bb.y;
294                        vMax = bb.y + bb.height;
295                        uCur = uMin;
296                        vCur = vMin;
297                        pNext = null;
298                }
299
300                /** 
301                 * Search from position (uCur, vCur) for the next valid region pixel.
302                 * Return the next position as a Point or null if no such point can be found.
303                 * Don't assume that (uCur, vCur) is a valid region pixel!
304                 * 
305                 * @return the next point
306                 */
307                private Point findNext() {
308                        // start search for next region pixel at (u,v):
309                        int u = (first) ? uCur : uCur + 1;
310                        int v = vCur;
311                        first = false;
312                        while (v <= vMax) {
313                                while (u <= uMax) {
314                                        if (getLabel(u, v) == label) { // next pixel found (uses surrounding labeling)
315                                                uCur = u;
316                                                vCur = v;
317                                                return new Point(uCur, vCur);
318                                        }
319                                        u++;
320                                }
321                                v++;
322                                u = uMin;
323                        }
324                        uCur = uMax + 1;        // just to make sure we'll never enter the loop again
325                        vCur = vMax + 1;
326                        return null;            // no next pixel found
327                }
328
329                @Override
330                public boolean hasNext() {
331                        if (pNext != null) {    // next element has been queried before but not consumed
332                                return true;
333                        }
334                        else {
335                                pNext = findNext();     // keep next pixel coordinates in pNext
336                                return (pNext != null);
337                        }
338                }
339
340                // Returns: the next element in the iteration
341                // Throws: NoSuchElementException - if the iteration has no more elements.
342                @Override
343                public Point next() {
344                        if (pNext != null || hasNext()) {
345                                Point pn = pNext;
346                                pNext = null;           // "consume" pNext
347                                return pn;
348                        }
349                        else {
350                                throw new NoSuchElementException();
351                        }
352                }
353
354                @Override
355                public void remove() {
356                        throw new UnsupportedOperationException();
357                }
358
359        } // end of RegionPixelIterator

362        /**
363         * This class of {@link RegionLabeling} represents a connected 
364         * component or binary region. 
365         * It is implemented as an inner class to {@link RegionLabeling} because
366         * it references common region labeling data.
367         * Instances of this class support iteration over the contained pixel
368         * coordinates of type {@link Point}, e.g., by
369         * <pre>
370         * import java.awt.Point;
371         * 
372         * BinaryRegion R = ...;
373         * for (Point p : R) {
374         *    // process point p ...
375         * }
376         * </pre>
377         */
378        public class BinaryRegion implements Comparable<BinaryRegion>, Iterable<Point> {
379                private final int label;                                // the label of THIS region
380                private int size = 0;
381                private double xc = Double.NaN;
382                private double yc = Double.NaN;
383                private int left = Integer.MAX_VALUE;
384                private int right = -1;
385                private int top = Integer.MAX_VALUE;
386                private int bottom = -1;
387                
388                private Contour outerContour;
389                private List<Contour> innerContours;
390                
391                // summation variables used for various statistics
392                private long x1Sum = 0;
393                private long y1Sum = 0;
394                private long x2Sum = 0;
395                private long y2Sum = 0;
396                private long xySum = 0;
397                
398                // ------- constructor --------------------------
399                
400                private BinaryRegion(int label) {
401                        this.label = label;
402                        this.outerContour = null;
403                        this.innerContours = null;
404                }
405                
406                /**
407                 * Get the x-value of the region's centroid.
408                 * @return the x-value of the region's centroid.
409                 */
410                public double getXc() {
411                        return xc;
412                }
413
414                /**
415                 * Get the y-value of the region's centroid.
416                 * @return the y-value of the region's centroid.
417                 */
418                public double getYc() {
419                        return yc;
420                }
421                
422                /**
423                 * Returns the sum of the x-coordinates of the points
424                 * contained in this region.
425                 * 
426                 * @return the sum of x-values.
427                 */
428                public long getX1Sum() {
429                        return x1Sum;
430                }
431
432                /**
433                 * Returns the sum of the y-coordinates of the points
434                 * contained in this region.
435                 * 
436                 * @return the sum of y-values.
437                 */
438                public long getY1Sum() {
439                        return y1Sum;
440                }
441
442                /**
443                 * Returns the sum of the squared x-coordinates of the points
444                 * contained in this region.
445                 * 
446                 * @return the sum of squared x-values.
447                 */
448                public long getX2Sum() {
449                        return x2Sum;
450                }
451
452                /**
453                 * Returns the sum of the squared y-coordinates of the points
454                 * contained in this region.

456                 * @return the sum of squared y-values.

458                public long getY2Sum() {
459                        return y2Sum;

461                
462                /**
463                 * Returns the sum of the mixed x*y-coordinates of the points
464                 * contained in this region.

466                 * @return the sum of xy-values.

468                public long getXYSum() {
469                        return xySum;
470                }

472                // ------- public methods --------------------------
473                
474                /**
475                 * Get the label associated with this region.
476                 * @return the region label.
477                 */
478                public int getLabel() {
479                        return this.label;
480                }
481                
482                /**
483                 * Get the size of this region.
484                 * @return the number of region points.
485                 */
486                public int getSize() {
487                        return this.size;
488                }
489                
490                /**
491                 * Get the x/y axes-parallel bounding box as a rectangle
492                 * (including the extremal coordinates).
493                 * @return the bounding box rectangle.
494                 */
495                public Rectangle getBoundingBox() {
496                        if (right < 0)
497                                return null;
498                        else
499                                return new Rectangle(left, top, right-left + 1, bottom - top + 1);
500                }

502                /**
503                 * @deprecated
504                 * Returns the centroid of this region as a 2D point.
505                 * Use {@link getCenterPoint} or {@link getXc} and {@link getYc} instead.
506                 * @return the centroid of this region.
507                 */
508                public Point2D getCenter() {
509                        if (Double.isNaN(xc))
510                                return null;
511                        else
512                                return new Point2D.Double(xc, yc);
513                }

517                 * Returns the centroid of this region as a 2D point.
518                 * See also {@link getXc}, {@link getYc}.
519                 * 
520                 * @return the centroid of this region.
521                 */
522                public Point2D getCenterPoint() {
523                        if (Double.isNaN(xc))
524                                return null;
525                        else
526                                return new Point2D.Double(xc, yc);
527                }

551                /**
552                 * Use this method to add a single pixel to this region. Updates summation
553                 * and boundary variables used to calculate various region statistics.
554                 * 
555                 * @param u x-position
556                 * @param v y-position
557                 */
558                protected void addPixel(int u, int v) {
559                        size = size + 1;
560                        x1Sum = x1Sum + u;
561                        y1Sum = y1Sum + v;
562                        x2Sum = x2Sum + u * u;
563                        y2Sum = y2Sum + v * v;
564                        xySum = xySum + u * v;
565                        if (u < left)   left = u;
566                        if (v < top)    top = v;
567                        if (u > right)  right = u;
568                        if (v > bottom) bottom = v;
569                }
570                
571                /**
572                 * Call this method to update the region's statistics. For now only the
573                 * center coordinates (xc, yc) are updated. Add additional statements as
574                 * needed to update your own region statistics.
575                 */
576                protected void update() {
577                        if (size > 0) {
578                                xc = (double) x1Sum / size;
579                                yc = (double) y1Sum / size;
580                        }
581                }
582                
583                /**
584                 * Get the (single) outer contour of this region.
585                 * Points on an outer contour are arranged in clockwise
586                 * order.
587                 * @return the outer contour.
588                 */
589                public Contour getOuterContour() {
590                        return outerContour;
591                }
592                
593                protected void setOuterContour(Contour contr) {
594                        outerContour = contr;
595                }
596                
597                /**
598                 * Get all inner contours of this region.
599                 * Points on inner contours are arranged in counter-clockwise
600                 * order.
601                 * @return the list of inner contours.
602                 */
603                public List<Contour> getInnerContours() {
604                        return innerContours;
605                }
606                
607                protected void makeInnerContours() {
608                        if (innerContours == null) {
609                                innerContours = new LinkedList<Contour>();
610                        }
611                }
612                
613                protected void addInnerContour(Contour contr) {
614                        makeInnerContours();
615                        innerContours.add(contr);
616                }
617                
618                public String toString() {
619                        Formatter fm = new Formatter(new StringBuilder(), Locale.US);
620                        fm.format("Region %d", label);
621                        fm.format(", area = %d", size);
622                        fm.format(", bounding box = (%d, %d, %d, %d)", left, top, right, bottom );
623                        fm.format(", centroid = (%.2f, %.2f)", xc, yc);
624                        if (innerContours != null)
625                                fm.format(", holes = %d", innerContours.size());
626                        String s = fm.toString();
627                        fm.close();
628                        return s;
629                }
630                
631                // Compare method for sorting by region size (larger regions at front)
632                @Override
633                public int compareTo(BinaryRegion r2) {
634                        return r2.size - this.size;
635                }

637                /* EXPERIMENTAL STUFF for attaching region properties dynamically:
638                 * Properties can be used to hash results of region calculations
639                 * to avoid multiple calculations.
640                 * Currently, only 'double' values are supported.
641                 * 
642                 * E.g. calculate major axis angle theta for region r, then do
643                 *    r.setProperty("angle", theta);
644                 * and subsequently
645                 *    double theta = r.getProperty("angle");
646                 */
647                
648                private Map<String, Double> properties = null;
649                
650                /**
651                 * Sets the specified property of this region to the given value.
652                 * @param name Chosen name of the property.
653                 * @param val Value associated with this property.
654                 */
655                public void setProperty(String name, double val) {
656                        if (properties == null) {
657                                properties = new HashMap<String, Double>();
658                        }
659                        properties.put(name, val);
660                }
661                
662                /**
663                 * Retrieves the specified region property. 
664                 * An {@link IllegalArgumentException} is thrown if the property 
665                 * is not defined for this region.
666                 * 
667                 * @param name The name of the property.
668                 * @return The value associated with the specified property.
669                 */
670                public double getProperty(String name) {
671                        Double value;
672                        if (properties == null || (value = properties.get(name)) == null) {
673                                throw new IllegalArgumentException("Region property " + name + " is undefined.");
674                        }
675                        return value.doubleValue();
676                }
677                
678                /**
679                 * Removes all properties attached to this region.
680                 */
681                public void clearProperties() {
682                        properties.clear();
683                }
