001package imagingbook.pub.geometry.basic;
002
003import static imagingbook.lib.math.Arithmetic.sqr;
004
005import java.util.Collection;
006
007/** 
008 * Interface specifying the behavior of a simple 2D point.
009 */
010public interface Point {
011        double getX();
012        double getY();
013        
014        static Point create(double x, double y) {
015                return new Imp(x, y);
016        }
017        
018        static Point create(double[] xy) {
019                return new Imp(xy[0], xy[1]);
020        }
021        
022        default double[] toArray() {
023                return new double[] {this.getX(), this.getY()};
024        }
025        
026        static Point[] toArray(Collection<Point> points) {
027                return points.toArray(new Point[0]);
028        }
029        
030        static double distance(Point p, Point q) {
031                return Math.sqrt(distance2(p, p));
032        }
033        
034        static double distance2(Point p, Point q) {
035                return sqr(q.getX() - p.getX()) + sqr(q.getY() - p.getY());
036        }
037        
038        default double distance2(Point other) {
039                return distance2(this, other);
040        }
041        
042        default double distance(Point other) {
043                return distance(this, other);
044        }
045        
046        
047        /**
048         * Simple fallback implementation of the {@link Point} interface.
049         * If dependency on AWT is to be avoided, just implement your own.
050         */
051        public class Imp extends java.awt.geom.Point2D.Double implements Point {
052                private static final long serialVersionUID = 1L;
053
054                public Imp(double x, double y) {
055                        super(x, y);
056                }
057                
058                @Override
059                public String toString() {
060                        return String.format("Point[%.3f, %.3f]", this.getX(), this.getY());
061                }
062        }
063
064}
