010package imagingbook.pub.geometry.mappings;
011
012import ij.process.ImageProcessor;
013import imagingbook.lib.image.ImageAccessor;
014import imagingbook.lib.interpolation.InterpolationMethod;
015
016import java.awt.Point;
017import java.awt.geom.Point2D;
018
019/**
020 * Abstract class which represents a general 2D transformation.
021 * @version 2019/01/01
022 */
023public abstract class Mapping implements Cloneable {
024        
025
026        /** 
027         * Duplicates this mapping by using {@link Object}'s shallow clone method.
028         * Sub-classes are supposed to override this method.
029         * @return a copy of this mapping
030         */
031        public Mapping duplicate() {
032                Mapping copy = null;
033                try {
034                        copy = (Mapping) super.clone();
035                } 
036                catch (CloneNotSupportedException e) {
037                        throw new UnsupportedOperationException("Cannot duplicate mapping " + this.toString());}
038                return copy;
039        }
040        
041        /**
042         * Applies this transformation to the given 2D point.
043         * All sub-classes of {@link Mapping} must implement this method.
044         * 
045         * @param x the x-coordinate of the point to be mapped
046         * @param y the y-coordinate of the point to be mapped
047         * @return the transformed 2D coordinate
048         */
049        public abstract double[] applyTo (double x, double y);
050        
051        /**
052         * Applies this transformation to a 2D point given as
053         * a {@code double} array.
054         * 
055         * @param xy the point to be mapped
056         * @return the transformed 2D coordinate
057         */
058        public double[] applyTo (double[] xy) {
059                return applyTo(xy[0], xy[1]);
060        }
061
062        /**
063         * The inverse of this mapping is calculated (if possible)
064         * and returned.
065         * 
066         * @return the inverse mapping
067         */
068        public Mapping getInverse() {
069                throw new UnsupportedOperationException("Cannot invert mapping " + this.toString());
070        }
071        
072        /**
073         * Applies this mapping to a single 2D point.
074         * 
075         * @param pnt the original point
076         * @return the transformed point
077         */
078        public Point2D applyTo(Point2D pnt) {
079                double[] xy = applyTo(new double[] {pnt.getX(), pnt.getY()});
080                return new Point2D.Double(xy[0], xy[1]);
081        }
082        
083        /**
084         * Applies this mapping to all points in the array of 2D points
085         * (convenience method).
086         * 
087         * @param pnts the original points
088         * @return the transformed points
089         */
090        public Point2D[] applyTo(Point2D[] pnts) {
091                Point2D[] outPnts = new Point2D[pnts.length];
092                for (int i = 0; i < pnts.length; i++) {
093                        outPnts[i] = applyTo(pnts[i]);
094                }
095                return outPnts;
096        }
097        
098        // image-to-image mappings (needs to be moved!) ----------------------
099
100        /**
101         * Destructively transforms the passed image using this geometric
102         * mapping and the specified pixel interpolation method.
103         * The geometric mapping is supposed to be INVERTED, i.e. transforming
104         * target to source image coordinates!
105         * TODO: this should not be here (geometry only)?
106         * 
107         * @param ip target image to which this mapping is applied
108         * @param im interpolation method
109         */
110        public void applyTo(ImageProcessor ip, InterpolationMethod im) {
111                // make a temporary copy of the image:
112                ImageProcessor source = ip.duplicate();
113                ImageProcessor target = ip;
114                applyTo(source, target, im);
115                source = null;
116        }
117
118        /**
119         * Transforms the source image to the target image using this geometric
120         * mapping and the specified pixel interpolation method. Note that source
121         * and target must be different images!
122         * The geometric mapping is supposed to be INVERTED, i.e. transforming
123         * target to source image coordinates!
124         * 
125         * @param source input image (not modified)
126         * @param target output image (modified)
127         * @param im interpolation method
128         */
129        public void applyTo(ImageProcessor source, ImageProcessor target, InterpolationMethod im) {
130                if (target == source) {
131                        throw new IllegalArgumentException("Source and target image must not be the same!");
132                }
133                ImageAccessor sourceAcc = ImageAccessor.create(source, null, im);
134                ImageAccessor targetAcc = ImageAccessor.create(target);
135                applyTo(sourceAcc, targetAcc);
136        }
137
138
139        /**
140         * Transforms the source image to the target image using this geometric
141         * mapping and the specified pixel interpolation method.
142         * The two images are passed as instances of {@link ImageAccessor}.
143         * Note that source and target reference different images!
144         * The geometric mapping is supposed to be INVERTED, i.e. transforming
145         * target to source image coordinates!
146         * 
147         * @param sourceAcc accessor to the source image
148         * @param targetAcc accessor to the target image
149         */
150        public void applyTo(ImageAccessor sourceAcc, ImageAccessor targetAcc) {
151                if (targetAcc.getProcessor() == sourceAcc.getProcessor()) {
152                        throw new IllegalArgumentException("Source and target image must not be the same!");
153                }
154                Mapping invMap = this;          // this always IS an inverse mapping!!
155                ImageProcessor target = targetAcc.getProcessor();
156                final int w = target.getWidth();
157                final int h = target.getHeight();
158                for (int v = 0; v < h; v++) {
159                        for (int u = 0; u < w; u++) {
160                                Point2D sourcePt = invMap.applyTo(new Point(u, v));
161                                float[] val = sourceAcc.getPix(sourcePt.getX(), sourcePt.getY());
162                                targetAcc.setPix(u, v, val);
163                        }
164                }
165        }
166}
