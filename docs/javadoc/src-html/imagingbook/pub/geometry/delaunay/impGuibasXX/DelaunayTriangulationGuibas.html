025        public static boolean SHUFFLE_INPUT_POINTS = true;

027        private final List<Vector2D> points;
028        private final List<Triangle2D> triangles;
029        private final Triangle2D superTriangle;
030
031        /**
032         * Constructor.
033         * @param pointSet the point set to be triangulated
034         * @param shuffle set {@code true} to randomly shuffle the input points
035         */
036        public DelaunayTriangulationGuibas(Collection<Point> pointSet, boolean shuffle) {
037                if (pointSet == null || pointSet.size() < 3) {
038                        throw new IllegalArgumentException("Point set must contain at least 3 points.");
039                }
040                this.triangles = new ArrayList<>();
041                this.superTriangle = new Triangle2D(Utils.makeOuterTriangle(pointSet));
042                this.points = makePointset(pointSet, shuffle);
043                triangulate();
044        }
045        
046        public DelaunayTriangulationGuibas(Collection<Point> pointSet) {
047                this(pointSet, SHUFFLE_INPUT_POINTS);
048        }

052        /** 
053         * Converts the incoming points (of unknown type but implementing the {@link Point} interface)
054         * to instances of the local implementation ({@link Vector2D}).
055         * 
056         * @param inPoints the input points (must implement {@link Point})
057         * @param shuffle if {@code true}, the input point sequence is randomly permuted
058         * @return the new point sequence
059         */
060        private List<Vector2D> makePointset(Collection<? extends Point> inPoints, boolean shuffle) {
061                Vector2D[] outPoints = new Vector2D[inPoints.size()];
062                int i = 0;
063                for (Point ip : inPoints) {
064                        outPoints[i] = new Vector2D(ip);
065                        i++;
066                }
067                List<Vector2D> outList = Arrays.asList(outPoints);
068                if (shuffle) {
069                        Collections.shuffle(outList);   // random permutation of pointset (in-place)
070                }
071                return outList;
072        }

074        private void triangulate() {
075//              final Triangle2D superTriangle = //this.makeOuterTriangle(points);
076//                              new Triangle2D(Utils.makeOuterTriangle(points));
077                                
078                triangles.add(superTriangle);
079                
080                for (Vector2D pnt : points) {
081                        Triangle2D triangle = findContainingTriangle(pnt);
082
083                        if (triangle == null) {
084                                /* If no containing triangle exists, then the vertex is not inside a triangle
085                                 * (this can also happen due to numerical errors) and lies on an edge. In order
086                                 * to find this edge we search all edges of the triangle soup and select the one
087                                 * which is nearest to the point we try to add. This edge is removed and four
088                                 * new edges are added.
089                                 */
090                                Edge2D edge = findNearestEdge(pnt);
091                                Triangle2D tr1 = findOneTriangleSharing(edge);
092                                Triangle2D tr2 = findNeighbour(tr1, edge);
093
094                                Vector2D oppositeVertex1 = tr1.getOppositeVertex(edge);
095                                Vector2D oppositeVertex2 = tr2.getOppositeVertex(edge);
096
097                                triangles.remove(tr1);
098                                triangles.remove(tr2);
099
100                                Triangle2D triangle1 = new Triangle2D(edge.a, oppositeVertex1, pnt);
101                                Triangle2D triangle2 = new Triangle2D(edge.b, oppositeVertex1, pnt);
102                                Triangle2D triangle3 = new Triangle2D(edge.a, oppositeVertex2, pnt);
103                                Triangle2D triangle4 = new Triangle2D(edge.b, oppositeVertex2, pnt);
104
105                                triangles.add(triangle1);
106                                triangles.add(triangle2);
107                                triangles.add(triangle3);
108                                triangles.add(triangle4);
109
110                                legalizeEdge(triangle1, new Edge2D(edge.a, oppositeVertex1), pnt);
111                                legalizeEdge(triangle2, new Edge2D(edge.b, oppositeVertex1), pnt);
112                                legalizeEdge(triangle3, new Edge2D(edge.a, oppositeVertex2), pnt);
113                                legalizeEdge(triangle4, new Edge2D(edge.b, oppositeVertex2), pnt);
114                        } 
115                        else { // pnt is inside the triangle <a,b,c>.
116                                Vector2D a = triangle.a;
117                                Vector2D b = triangle.b;
118                                Vector2D c = triangle.c;
119
120                                triangles.remove(triangle);
121
122                                Triangle2D triangle1 = new Triangle2D(a, b, pnt);
123                                Triangle2D triangle2 = new Triangle2D(b, c, pnt);
124                                Triangle2D triangle3 = new Triangle2D(c, a, pnt);
125
126                                triangles.add(triangle1);
127                                triangles.add(triangle2);
128                                triangles.add(triangle3);
129
130                                legalizeEdge(triangle1, new Edge2D(a, b), pnt);
131                                legalizeEdge(triangle2, new Edge2D(b, c), pnt);
132                                legalizeEdge(triangle3, new Edge2D(c, a), pnt);
133                        }
134                }
135
136                // Remove all triangles that contain vertices of the super triangle:
137                removeTrianglesUsing(superTriangle.a);
138                removeTrianglesUsing(superTriangle.b);
139                removeTrianglesUsing(superTriangle.c);
140        }

163        /**
164         * This method legalizes edges by recursively flipping all illegal edges.
165         * 
166         * @param triangle  the triangle
167         * @param edge      the edge to be legalized
168         * @param newVertex the new vertex
169         */
170        private void legalizeEdge(Triangle2D triangle, Edge2D edge, Vector2D newVertex) {
171                Triangle2D neighbourTriangle = findNeighbour(triangle, edge);
172
173                // If the triangle has a neighbor, then legalize the edge
174                if (neighbourTriangle != null) {
175                        if (neighbourTriangle.isPointInCircumCircle(newVertex)) {
176                                triangles.remove(triangle);
177                                triangles.remove(neighbourTriangle);
178
179                                Vector2D noneEdgeVertex = neighbourTriangle.getOppositeVertex(edge);
180
181                                Triangle2D triangle1 = new Triangle2D(noneEdgeVertex, edge.a, newVertex);
182                                Triangle2D triangle2 = new Triangle2D(noneEdgeVertex, edge.b, newVertex);
183
184                                triangles.add(triangle1);
185                                triangles.add(triangle2);
186
187                                legalizeEdge(triangle1, new Edge2D(noneEdgeVertex, edge.a), newVertex);
188                                legalizeEdge(triangle2, new Edge2D(noneEdgeVertex, edge.b), newVertex);
189                        }
190                }
191        }

193        /**
194         * Returns the triangles of the triangulation in form of a vector of 2D
195         * triangles. The initial 'superTriangle' is removed.
196         * The resulting triangles should form a convex structure.
197         * 
198         * @return the triangles of this triangulation.
199         */
200        @Override
201        public Collection<Triangle> getTriangles() {
202                return Collections.unmodifiableList(triangles);
203        }

206        public Collection<Point> getPoints() {
207                return Collections.unmodifiableList(points);

212        /**
213         * Returns the triangle that contains the specified point or null if no
214         * such triangle exists.
215         * @param point the query point
216         * @return the containing triangle or {@code null} if none was found
217         */
218        private Triangle2D findContainingTriangle(Vector2D point) {
219                for (Triangle2D triangle : triangles) {
220                        if (triangle.containsPoint(point)) {
221                                return triangle;
222                        }
223                }
224                return null;
225        }
226        
227        /**
228         * Returns the triangle edge nearest to the specified point.
229         * wilbur: modified massively (no lists, array, no sorting)
230         * @param point the query point
231         * @return the triangle edge nearest to the specified point
232         */
233        private Edge2D findNearestEdge(Vector2D point) {
234                Edge2D minEdge = null;
235                double minDist = Double.POSITIVE_INFINITY;
236                for (Triangle2D tri : triangles) {
237                        Distance ed = tri.findNearestEdge(point);
238                        double dist = ed.getDistance();
239                        if (dist < minDist) {
240                                minDist = dist;
241                                minEdge = ed.getEdge();
242                        }
243                }
244                return minEdge;
245        }
246        
247        /**
248         * Returns one of the possible triangles sharing the specified edge. Based on
249         * the ordering of the triangles in this triangle soup the returned triangle may
250         * differ. To find the other triangle that shares this edge use the
251         * {@link findNeighbour(Triangle2D triangle, Edge2D edge)} method.
252         * @param edge The edge
253         * @return Returns one triangle that shares the specified edge
254         */
255        private Triangle2D findOneTriangleSharing(Edge2D edge) {
256                for (Triangle2D triangle : triangles) {
257                        if (triangle.containsEdge(edge)) {
258                                return triangle;
259                        }
260                }
261                return null;
262        }
263        
264        /**
265         * Returns the neighbor triangle of the specified triangle sharing the same edge
266         * as specified. If no neighbor sharing the same edge exists null is returned.
267         * TODO: searching over ALL triangles seems to be unnecessarily expensive
268         * 
269         * @param tri1 The triangle
270         * @param edge     The edge
271         * @return The triangles neighbor triangle sharing the same edge or null if no
272         *         triangle exists
273         */
274        private Triangle2D findNeighbour(Triangle2D tri1, Edge2D edge) {
275                for (Triangle2D tri2 : triangles) {
276                        if (tri2.containsEdge(edge) && tri2 != tri1) {
277                                return tri2;
278                        }
279                }
280                return null;
281        }
282        
283        /**
284         * Removes all triangles that contain the specified corner point.
285         * @param point the corner point
286         */
287        private void removeTrianglesUsing(Vector2D point) {
288                List<Triangle2D> trianglesToBeRemoved = new LinkedList<>();
289                for (Triangle2D triangle : triangles) {
290                        if (triangle.hasVertex(point)) {
291                                trianglesToBeRemoved.add(triangle);
292                        }
293                }
294                triangles.removeAll(trianglesToBeRemoved);
295        }
