010package imagingbook.pub.color.filters;
011
012import ij.ImagePlus;
013import ij.process.FloatProcessor;
014import ij.process.ImageProcessor;
015
016public class FilterMask {
017        
018        private final int center;                       // mask center position
019        private final int count;                        // number of nonzero mask elements
020        private final int[][] mask;                     // mask[x][y]  specifies the support region
021                
022        public FilterMask(double radius) {
023                center = Math.max((int) Math.ceil(radius), 1);
024                //IJ.log("mask radius = " + center);
025                int mWidth = 2 * center + 1;            // width/height of mask array
026                mask = new int[mWidth][mWidth];                 // initialized to zero
027                int cnt = 0;
028                double r2 = radius * radius + 1;                // add 1 to get mask shape similar to ImageJ
029                for (int u = 0; u < mWidth; u++) {
030                        int x = u - center;
031                        for (int v = 0; v < mWidth; v++) {
032                                int y = v - center;
033                                if (x*x + y*y <= r2) {
034                                        mask[u][v] = 1;
035                                        cnt = cnt + 1;
036                                }
037                        }
038                }
039                count = cnt;
040        }
041        
042        public int getCenter() {
043                return center;
044        }
045        
046        public int getCount() {
047                return count;
048        }
049        
050        public int[][] getMask() {
051                return mask;
052        }
053        
054        public ImagePlus getImagePlus(String title) {
055                FloatProcessor fp = new FloatProcessor(mask);
056                ImageProcessor bp = fp.convertToByte(false);
057                bp.threshold(0);
058                //bp.multiply(255);
059                return new ImagePlus(title, bp);
060        }
061        
062        public void show(String title) {
063                getImagePlus(title).show();
064        }
065
066}
