023/**
024 * This class implements a Canny edge detector for grayscale and RGB images.
025 * The edge detector is "lazy" in the sense that it performs local non-
026 * maximum suppression and edge tracing only when the results are explicitly 
027 * asked for (by the methods {@code getEdgeBinary()} and {@code getEdgeTraces()}).
028 * 
029 * @author W. Burger
030 * @version 2017/04/13
031 */
032public class CannyEdgeDetector extends ColorEdgeDetector {
033        
034        public static class Parameters {
035                /** Gaussian sigma (scale) */
036                public double gSigma = 2.0f;
037                
038                /** High threshold (20% of max. edge magnitude) */
039                public double hiThr  = 20.0f;
040                
041                /** Low threshold (5% of max. edge magnitude) */
042                public double loThr = 5.0f;
043                
044                /** Set true to normalize gradient magnitude */
045                public boolean normGradMag = true;
046                
047                /**
048                 * Checks the parameter set.
049                 * @return true if any invalid condition is found
050                 */
051                public boolean isInValid () { 
052                        return gSigma < 0.1f || loThr > hiThr;
053                }
054        }

056        private final Parameters params;
057//      private final ImageProcessor I;                         // original image
058        private final int M, N;                                         // width and height of I
059        
060        private FloatProcessor Emag = null;                             // gradient magnitude
061        private FloatProcessor Enms = null;                             // non-max suppressed gradient magnitude
062        private FloatProcessor Ex = null, Ey = null;    // edge normal vectors
063        private ByteProcessor Ebin = null;                              // final (binary) edge image
064        private List<List<Point>> traceList = null;             // list of edge traces

072        public CannyEdgeDetector(ImageProcessor I, Parameters params) {

074                this.params = params;
075//              this.I = I;
076                M = I.getWidth();
077                N = I.getHeight();
078                makeGradientsAndMagnitude(I);
079        }

094        private void makeGradientsAndMagnitudeGray(ImageProcessor I) {          
095                FloatProcessor If = I.convertToFloatProcessor();        // always makes a copy
096                                
097                // apply a separable Gaussian filter to I
098                float[] gaussKernel = makeGaussKernel1d(params.gSigma);
099                Convolver conv = new Convolver();
100                conv.setNormalize(true);
101                conv.convolve(If, gaussKernel, gaussKernel.length, 1);
102                conv.convolve(If, gaussKernel, 1, gaussKernel.length);
103                
104                // calculate the gradients in X- and Y-direction
105                Ex = If;
106                Ey = (FloatProcessor) If.duplicate();
107                float[] gradKernel = {-0.5f, 0, 0.5f};
108                conv.setNormalize(false);
109                conv.convolve(Ex, gradKernel, gradKernel.length, 1);
110                conv.convolve(Ey, gradKernel, 1, gradKernel.length);
111                
112                Emag = new FloatProcessor(M, N);
113                float emax = 0;
114                for (int v = 0; v < N; v++) {
115                        for (int u = 0; u < M; u++) {
116                                float dx = Ex.getf(u,v);
117                                float dy = Ey.getf(u,v);
118                                float mag = (float) Math.hypot(dx, dy); // = (float) Math.sqrt(dx*dx + dy*dy);
119                                if (mag > emax) 
120                                        emax = mag;
121                                Emag.setf(u, v, mag);
122                        }
123                }
124                
125                // normalize gradient magnitude 
126                if (params.normGradMag && emax > 0.001) 
127                        Emag.multiply(100.0/emax);
128        }

130        private void makeGradientsAndMagnitudeColor(ColorProcessor I) {
131                FloatProcessor[] Irgb = rgbToFloatChannels(I);
132                FloatProcessor[] Ixrgb = new FloatProcessor[3];
133                FloatProcessor[] Iyrgb = new FloatProcessor[3];
134                
135                // apply a separable Gaussian filter to each RGB channel
136                float[] gaussKernel = makeGaussKernel1d(params.gSigma);
137                Convolver conv = new Convolver();
138                conv.setNormalize(true);
139                for (int i=0; i<Irgb.length; i++) {
140                        FloatProcessor If = Irgb[i];
141                        conv.convolve(If, gaussKernel, gaussKernel.length, 1);
142                        conv.convolve(If, gaussKernel, 1, gaussKernel.length);
143                        Ixrgb[i] = If;
144                        Iyrgb[i] = (FloatProcessor) If.duplicate();
145                }
146                
147                // calculate the gradients in X- and Y-direction for each RGB channel
148                float[] gradKernel = {-0.5f, 0, 0.5f};
149                conv.setNormalize(false);
150                for (int i = 0; i < Irgb.length; i++) {
151                        FloatProcessor Ix = Ixrgb[i];
152                        FloatProcessor Iy = Iyrgb[i];
153                        conv.convolve(Ix, gradKernel, gradKernel.length, 1);
154                        conv.convolve(Iy, gradKernel, 1, gradKernel.length);
155                }
156
157                // calculate gradient magnitude
158                Ex = new FloatProcessor(M, N);
159                Ey = new FloatProcessor(M, N);
160                Emag = new FloatProcessor(M, N);
161                float emax = 0;
162                for (int v = 0; v < N; v++) {
163                        for (int u = 0; u < M; u++) {
164                                float rx = Ixrgb[0].getf(u,v), ry = Iyrgb[0].getf(u,v);
165                                float gx = Ixrgb[1].getf(u,v), gy = Iyrgb[1].getf(u,v);
166                                float bx = Ixrgb[2].getf(u,v), by = Iyrgb[2].getf(u,v);
167                                float A = rx*rx + gx*gx + bx*bx;
168                                float B = ry*ry + gy*gy + by*by;
169                                float C = rx*ry + gx*gy + bx*by;
170                                float D = (float) Math.sqrt((A - B)*(A - B) + 4*C*C);
171                                
172                                float mag = (float) Math.sqrt(0.5*(A+B+D));
173                                if (mag > emax) emax = mag;
174                                Emag.setf(u, v, mag);
175                                Ex.setf(u, v, A - B + D);
176                                Ey.setf(u, v, 2*C);
177                        }
178                }
179                //IJ.log("RGB emax = " + emax);
180                // normalize gradient magnitude 
181                if (params.normGradMag && emax > 0.001) 
182                        Emag.multiply(100.0/emax);
183        }

188        private void nonMaxSuppression() {
189                // perform non-maximum suppression along gradient direction
190                Enms = new FloatProcessor(M, N);
191                for (int v = 1; v < N - 1; v++) {
192                        for (int u = 1; u < M - 1; u++) {
193                                int s_theta = getOrientationSector(Ex.getf(u, v), Ey.getf(u, v));
194                                if (isLocalMaximum(Emag, u, v, s_theta, (float) params.loThr)) {
195                                        Enms.setf(u, v, Emag.getf(u, v)); // keep local maximum only
196                                }
197                        }
198                }
199        }
200        
201        private void detectAndTraceEdges() {
202                if (Enms == null) {
203                        nonMaxSuppression();
204                }
205                Ebin = new ByteProcessor(M, N);
206                int color = 255;
207                traceList = new LinkedList<List<Point>>();
208                for (int v = 0; v < N; v++) {
209                        for (int u = 0; u < M; u++) {
210                                if (Enms.getf(u, v) >= params.hiThr && Ebin.get(u, v) == 0) { // unmarked edge point
211                                        List<Point> trace = traceAndThreshold(u, v, (float) params.loThr, color);
212                                        traceList.add(trace);
213                                }
214                        }
215                }
216        }

218        // Determines if the gradient magnitude is a local maximum at position (u,v)
219        // in direction s_theta.
220        private boolean isLocalMaximum(FloatProcessor gradMagnitude, int u, int v, int s_theta, float mMin) {
221                float mC = gradMagnitude.getf(u, v);
222                if (mC < mMin) {
223                        return false;
224                }
225                else {
226                        float mL = 0, mR = 0;
227                        switch (s_theta) {
228                        case 0 : 
229                                mL = gradMagnitude.getf(u-1, v);
230                                mR = gradMagnitude.getf(u+1, v);
231                                break;
232                        case 1 : 
233                                mL = gradMagnitude.getf(u-1, v-1);
234                                mR = gradMagnitude.getPixelValue(u+1, v+1);
235                                break;
236                        case 2 : 
237                                mL = gradMagnitude.getf(u, v-1);
238                                mR = gradMagnitude.getf(u, v+1);
239                                break;
240                        case 3 : 
241                                mL = gradMagnitude.getf(u-1, v+1);
242                                mR = gradMagnitude.getf(u+1, v-1);
243                                break;
244                        }
245                        return (mL <= mC && mC >= mR);
246                }
247        }

249        ///Recursively collect and mark all pixels of an edge that are 8-connected to 
250        // (u0,v0) and have a gradient magnitude above loThr.
251        /**
252         * Recursively collect and mark all pixels of an edge that are 8-connected to 
253         * (u0,v0) and have a gradient magnitude above loThr.
254         * @param u0 start coordinate (x)
255         * @param v0 start coordinate (y)
256         * @param loThr low threshold (min. edge magnitude to continue tracing)
257         * @param markColor color used for marking pixels on edge trace
258         * @return a list of Point objects.
259         */
260        private List<Point> traceAndThreshold(int u0, int v0, float loThr, int markColor) {
261                //IJ.log("  working point " + u + " " + v);
262                Stack<Point> pointStack = new Stack<Point>();
263                List<Point> pointList = new LinkedList<Point>();
264                pointStack.push(new Point(u0, v0));
265                while (!pointStack.isEmpty()) {
266                        Point p = pointStack.pop();
267                        int up = p.x;
268                        int vp = p.y;
269                        Ebin.putPixel(up, vp, markColor);       // mark this edge point
270                        pointList.add(p);
271                                
272                        int uL = Math.max(up - 1, 0);           // (up > 0) ? up-1 : 0;
273                        int uR = Math.min(up + 1, M - 1);       // (up < M-1) ? up+1 : M-1;
274                        int vT = Math.max(vp - 1, 0);           // (vp > 0) ? vp - 1 : 0;
275                        int vB = Math.min(vp + 1, N - 1);       // (vp < N-1) ? vp+1 : N-1;
276                        
277                        for (int u = uL; u <= uR; u++) {
278                                for (int v = vT; v <= vB; v++) {
279                                        if (Ebin.get(u, v) == 0 && Enms.getf(u, v) >= loThr) {
280                                                pointStack.push(new Point(u,v));
281                                        }
282                                }
283                        }
284                }
285                return pointList;
286        }

288        private final float cosPi8 = (float) Math.cos(Math.PI/8);
289        private final float sinPi8 = (float) Math.sin(Math.PI/8);
290        
291        // returns the quantized orientation sector for vector (dx, dy)
292        private int getOrientationSector(float dx, float dy) {
293                // rotate the gradient vector by PI/8
294                float dxR = cosPi8 * dx - sinPi8 * dy;
295                float dyR = sinPi8 * dx + cosPi8 * dy;  
296                // mirror vector (dxR,dyR) to [0,PI]
297                if (dyR < 0) {
298                        dxR = -dxR;
299                        dyR = -dyR;
300                }
301                // determine the octant for (dx, dy)
302                int s_theta;
303                if (dxR >= 0) { // dxR >= 0, dyR >= 0
304                        if (dxR >= dyR)
305                                s_theta = 0; // return 0;
306                        else
307                                s_theta = 1; // return 1;
308                }
309                else {  // dxR < 0, dyR >= 0
310                        if (-dxR < dyR)
311                                s_theta = 2; // return 2;
312                        else
313                                s_theta = 3; //return 3;
314                }
315                return s_theta;
316        }

322        public FloatProcessor getEdgeOrientation() {
323                FloatProcessor E_theta = new FloatProcessor(M, N);
324                for (int u = 0; u < M; u++) {
325                        for (int v = 0; v < N; v++) {
326                                double ex = Ex.getf(u, v);
327                                double ey = Ey.getf(u, v);
328                                float theta = (float) Math.atan2(ey, ex);
329                                E_theta.setf(u, v, theta);
330                        }
331                }
332                return E_theta;
333        }

351        private float[] makeGaussKernel1d(double sigma) {
352                // make 1D Gauss filter kernel large enough
353                int rad = (int) (3.5 * sigma);
354                int size = rad + rad + 1;
355                float[] kernel = new float[size]; // center cell = kernel[rad]
356                double sigma2 = sigma * sigma;
357                double scale = 1 / Math.sqrt(2 * Math.PI * sigma2);
358                for (int i = 0; i < size; i++) {
359                        double x = rad - i;
360                        kernel[i] = (float) (
361                        scale * Math.exp(-0.5 * (x * x) / sigma2)
362                        );
363                }
364                return kernel;
365        }

367        // extract RGB channels of 'cp' as 3 float processors
368        private FloatProcessor[] rgbToFloatChannels(ColorProcessor cp) {
369                int w = cp.getWidth();
370                int h = cp.getHeight();
371                FloatProcessor rp = new FloatProcessor(w, h);
372                FloatProcessor gp = new FloatProcessor(w, h);
373                FloatProcessor bp = new FloatProcessor(w, h);
374                int[] pixels = (int[]) cp.getPixels(); 
375                float[] rpix = (float[]) rp.getPixels();
376                float[] gpix = (float[]) gp.getPixels();
377                float[] bpix = (float[]) bp.getPixels();
378        for (int i=0; i<pixels.length; i++) {
379            int c = pixels[i];
380            rpix[i] = (c&0xff0000)>>16;
381                gpix[i] = (c&0xff00)>>8;
382                        bpix[i] =  c&0xff;
383        }
384        return new FloatProcessor[] {rp, gp, bp};
385        }
