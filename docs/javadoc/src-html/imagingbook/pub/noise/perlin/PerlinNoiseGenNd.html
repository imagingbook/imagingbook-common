010package imagingbook.pub.noise.perlin;
011
012import imagingbook.lib.math.Matrix;
013import imagingbook.lib.math.Matrix.IncompatibleDimensionsException;
014import imagingbook.pub.noise.hashing.HashFun;
015
016/**
017 * Gradient (Perlin) noise implementation.
018 * This class implements an N-dimensional Perlin noise generator.
019 */
020public class PerlinNoiseGenNd extends PerlinNoiseGen {
021        
022        final int N;            // dimensionality, default 1
023        final int K;                    // number of hypercube vertices, default 2
024        final int[][] Q;                        // vertex coordinates of the unit hypercube
025        
026        public PerlinNoiseGenNd(int N, double f_min, double f_max, double persistence, HashFun hf) {
027                super(f_min, f_max, persistence, hf);
028                this.N = N;
029                this.K = (int) Math.pow(2, N);  // number of hypercube vertices
030                this.Q = new int[K][N];                 // vertices of the unit hypercube
031                for (int j = 0; j < K; j++) {
032                        Q[j] = vertex(j, N);
033                }
034        }
035        
036        /**
037         * N-dim combined (multi-frequency) Perlin noise function. 
038         * @param X Interpolation position X (N-dimensional).
039         * @return The value of the combined Perlin
040         * noise function for the N-dimensional position X.
041         */
042        public double NOISE(double[] X) {
043                double sum = 0;
044                for (int i = 0; i < F.length; i++) { // for all frequencies
045                        sum = sum + A[i] * noise(Matrix.multiply(F[i], X));
046                }
047                return sum;
048        }
049        
050        /**
051         * 2D elementary (single-frequency) Perlin noise function. 
052         * @param X Interpolation position X (N-dimensional).
053         * @return The value of the elementary Perlin
054         * noise function for the N-dimensional position X.
055         */
056        public double noise(double[] X) {
057                int[] P0 = floor(X);            // origin of hypercube around X
058                 
059                // get the 2^N gradient vectors for all hypercube corners:
060                double[][] G = new double[K][N];
061                for(int j=0; j<K; j++) {        
062                        G[j] = gradient(add(P0,Q[j]));                  // gradient vector at cube corner j
063                }
064                
065                double[] X01 = subtract(X,P0);                                  // X01[k] are in [0,1]
066                
067                // get the 2^N gradient values at all vertices for position X
068                double[] W = new double[K];
069                for (int j = 0; j < K; j++) {   
070                        W[j] = Matrix.dotProduct(G[j], subtract(X01, Q[j]));
071                }
072                
073                return interpolate(X01, W, 0);
074        }
075        
076        /**
077         * @param p discrete position.
078         * @return A pseudo-random gradient vector for 
079         * the discrete lattice point p (N-dimensional).
080         */
081        double[] gradient(int[] p) {    
082                if (p.length == 2) {
083                        return gradient(p[0],p[1]);
084                }
085                // hash() always returns a new double[], g[i] in [0,1]
086                double[] g = hashFun.hash(p);   // STILL TO BE DONE!!!
087                for (int i=0; i<g.length; i++) {
088                        g[i] = 2.0 * g[i] - 1;
089                }
090                return g;
091        }
092        
093        /**
094         * Local interpolation function.
095         * @param X01 Interpolation position in [0,1]^N
096         * @param WW  A vector of length 2^(N-d) with
097         * the tangent values for the hypercube corners.
098         * @param k The interpolation dimension (axis).
099         * @return  The interpolated noise value at position X01.
100         */
101        double interpolate(double[] X01, double[] WW, int k) { 
102                if (WW.length == 1) {                   // (d == N)
103                        return WW[0];                           // done, end of recursion
104                }
105                else {                                                  // d < N
106                        double x01 = X01[k];            // select dimension d of vector X
107                        double s = this.s(x01);                 // blending function
108                        int M = WW.length/2;
109                        double[] W = new double[M];             // W is half the length of WW
110                        for (int i=0; i<M; i++) {
111                                double wa = WW[2*i];
112                                double wb = WW[2*i+1];
113                                W[i] = (1-s)*wa + s*wb;         // the actual interpolation
114                        }
115                        return interpolate(X01, W, k+1);
116                }
117        }
118
119        /**
120         * @param j Vertex number (0..2^N-1)
121         * @param N Dimension of the hypercube
122         * @return The coordinate vector for vertex j of the N-dimensional
123         * hypercube.
124         */
125        private int[] vertex(int j, int N) { 
126                int[] v = new int[N];
127                // copy the bit representation of j into v,
128                // v[0] is the most significant bit 
129                for (int k = 0; k < N; k++) {
130                         v[k] = j & 0x00000001;         // select least significant bit (bit 0)
131                         j = j >>> 1;                           // j <- j/2
132                }
133                return v;
134        }
135        
136        // from 2D example
137        double[] gradient(int i, int j) {
138                double[] g = hashFun.hash(i,j);         // hash() always returns a new double[]
139                g[0] = 2.0 * g[0] - 1;
140                g[1] = 2.0 * g[1] - 1;
141                return g;
142        }
143        
144        double[] subtract(double[] a, int[] b) {
145                if (a.length != b.length)
146                        throw new IncompatibleDimensionsException();
147                final int n = a.length;
148                double[] c = new double[n];
149                for (int i = 0; i < n; i++) {
150                        c[i] = a[i] - b[i];
151                }
152                return c;
153        }
154        
155        int[] add(int[] a, int[] b) {
156                if (a.length != b.length)
157                        throw new IncompatibleDimensionsException();
158                final int n = a.length;
159                int[] c = new int[n];
160                for (int i = 0; i < n; i++) {
161                        c[i] = c[i] + b[i];
162                }
163                return c;
164        }
165        
166        int[] floor(double[] a) {
167                int[] b = new int[a.length];
168                for (int i = 0; i < a.length; i++) {
169                        b[i] = (int) Math.floor(a[i]);
170                }
171                return b;
172        }
