010package imagingbook.pub.corners;
011
012import ij.process.ImageProcessor;
013import imagingbook.pub.geometry.basic.Point;
014
015/**
016 * This class represents a 2D corner.
017 * A corner is essentially a {@link Point} plus a scalar quantity
018 * {@link #q} for the corner strength.
019 * 
020 * @version 2020/02/25
021 */
022public class Corner implements Point, Comparable<Corner> {
023        protected final float x, y, q;
024
025        public double getX() {
026                return x;
027        }
028
029        public double getY() {
030                return y;
031        }
032
033        public double getQ() {
034                return q;
035        }
036        
037        public Corner (float x, float y, float q) {
038                this.x = x;
039                this.y = y;
040                this.q = q;
041        }
042        
043        double dist2 (Corner c2){
044                //returns the squared distance between this corner and corner c2
045                float dx = this.x - c2.x;
046                float dy = this.y - c2.y;
047                return (dx * dx) + (dy * dy);   
048        }
049
050        // used for sorting corners by corner strength q
051        @Override
052        public int compareTo (Corner c2) {
053                return Double.compare(c2.q, this.q);
054        }
055        
056        // ----------------------------------------------------------------
057        
058        public void draw(ImageProcessor ip, int size) {
059                int x = (int) Math.round(this.getX());
060                int y = (int) Math.round(this.getY());
061                ip.drawLine(x - size, y, x + size, y);
062                ip.drawLine(x, y - size, x, y + size);
063        }
064        
065}
