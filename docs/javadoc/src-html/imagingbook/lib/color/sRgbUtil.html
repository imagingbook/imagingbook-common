013/*
014 * This is only a utility class to hold static methods used by CIELAB and CIELUV color spaces.
015 * Should be modified to implement a a subclass of ColorSpace!
016 * TODO: duplicate in pub/color/image?? Add add JavaDoc!
017 */
018
019public abstract class sRgbUtil {
020        
021        // double versions of Gamma correction
022        
023    public static double gammaFwd(double lc) {  // input: linear component value
024                return (lc > 0.0031308) ?
025                        (1.055 * Math.pow(lc, 1/2.4) - 0.055) :
026                        (lc * 12.92);
027    }
028    
029    public static double gammaInv(double nc) {  // input: nonlinear component value
030        return (nc > 0.03928) ?
031                        Math.pow((nc + 0.055)/1.055, 2.4) :
032                        (nc / 12.92);
033    }
034    
035    public static float[] sRgbToRgb(float[] srgb) {     // all components in [0,1]
036                float R = (float) sRgbUtil.gammaInv(srgb[0]);
037                float G = (float) sRgbUtil.gammaInv(srgb[1]);
038                float B = (float) sRgbUtil.gammaInv(srgb[2]);
039        return new float[] {R,G,B};
040    }
041
042    public static float[] rgbToSrgb(float[] rgb) {      // all components in [0,1]
043                float sR = (float) sRgbUtil.gammaFwd(rgb[0]);
044                float sG = (float) sRgbUtil.gammaFwd(rgb[1]);
045                float sB = (float) sRgbUtil.gammaFwd(rgb[2]);
046                return new float[] {sR,sG,sB};
047    }
048    
049}
