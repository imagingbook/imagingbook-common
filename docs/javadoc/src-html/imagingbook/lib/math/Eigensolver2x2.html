013/**
014 * The this class computes the eigenvalues/eigenvectors for a 2x2
015 * matrix
016 * <pre>
017 *   | A B |
018 *   | C D | </pre>
019 * The implementation is inspired by Blinn, Jim: Consider the lowly 2x2 matrix. 
020 * IEEE Computer Graphics and Applications, 16(2):82-88, 1996.
021 * @author W. Burger
022 * @version 2013-07-29
023 */
024public class Eigensolver2x2 {
025        
026        private final double A, B, C, D;
027        
028        public Eigensolver2x2(double[][] A) {
029                this(A[0][0], A[0][1], A[1][0], A[1][1]);
030        }
031        
032        public Eigensolver2x2(double A, double B, double C, double D) {
033                this.A = A;
034                this.B = B;
035                this.C = C;
036                this.D = D;
037        }
038
039        public EigenPair[] realEigenValues2x2() {
040                final double R = (A + D) / 2;
041                final double S = (A - D) / 2;
042                if ((S * S + B * C) < 0) // no real eigenvalues
043                        return null;
044                else {
045                        double T = Math.sqrt(S * S + B * C);
046                        double eVal1 = R + T;
047                        double eVal2 = R - T;
048                        double[] eVec1, eVec2;
049                        if ((A - D) >= 0) {
050                                eVec1 = new double[] {S + T, C};
051                                eVec2 = new double[] {B, -S - T};
052                        } else {
053                                eVec1 = new double[] {B, -S + T};
054                                eVec2 = new double[] {S - T, C};
055                        }
056
057                        EigenPair[] e = new EigenPair[2];
058                        // put eigenpair with larger eigenvalue up front
059                        if (Math.abs(eVal1) >= Math.abs(eVal2)) {
060                                e[0] = new EigenPair(eVal1, eVec1);
061                                e[1] = new EigenPair(eVal2, eVec2);
062                        } else {
063                                e[1] = new EigenPair(eVal1, eVec1);
064                                e[0] = new EigenPair(eVal2, eVec2);
065                        }
066                        return e;
067                }
068        }
069
070        /**
071         * EigenPair is a tuple (eigenvalue, eigenvector) and represents
072         * the solution to an eigen problem.
073         */
074        public static class EigenPair {
075                final double eival;
076                final double[] eivec;
077
078                public EigenPair(double eival, double[] eivec) {
079                        this.eival = eival;
080                        this.eivec = eivec.clone();
081                }
082
083                public double getEigenvalue() {
084                        return this.eival;
085                }
086
087                public double[] getEigenvector() {
088                        return this.eivec;
089                }
090                
091                public String toString() {
092                        if (eivec == null)
093                                return "no eigenvalue / eigenvector";
094                        else {
095                                return String.format("eigenvalue: %.5f | eigenvector: %s", eival, Matrix.toString(eivec)) ;
096                        }
097                }
098        }
