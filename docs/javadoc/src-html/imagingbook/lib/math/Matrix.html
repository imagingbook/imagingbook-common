027/**
028 * This class contains a collection of static methods for calculations
029 * with vectors and matrices using native Java arrays without any enclosing 
030 * objects structures. 
031 * Matrices are simply two-dimensional array M[r][c], where r is the row index
032 * and c is the column index (as common in linear algebra). This means that
033 * matrices are really vectors of row vectors.
034 * 
035 * Methods named with a trailing 'D' (e.g., 'multiplyD') operate destructively,
036 * i.e., modify one of the passed arguments.
037 * 
038 * TODO: add JavaDoc comments.
039 * 
040 * @author WB
041 * @version 2017/02/20
042 */

300        /*  
301         * Implements a right (post-) matrix-vector multiplication: x . A -> y
302         * x is treated as a row vector of length m, matrix A is of size (m,n).
303         * y (a row vector of length n) is modified.
304         * destructive
305         */
306        public static void multiplyD(final double[] x, final double[][] A, double[] y) {
307                final int m = getNumberOfRows(A);
308                final int n = getNumberOfColumns(A);
309                if (x.length != m || y.length != n) 
310                        throw new IllegalArgumentException("incompatible vector-matrix dimensions");
311                for (int i = 0; i < n; i++) {
312                        double s = 0;
313                        for (int j = 0; j < m; j++) {
314                                s = s + x[j] * A[j][i];
315                        }
316                        y[i] = s;
317                }
318        }

393                final int nB = getNumberOfColumns(B);

402                }

415        // A * B -> C (destructive)
416        public static void multiply(final float[][] A, final float[][] B, final float[][] C) {
417                final int mA = getNumberOfRows(A);
418                final int nA = getNumberOfColumns(A);
419                final int mB = getNumberOfRows(B);
420                final int nB = getNumberOfColumns(B);
421                if (nA != mB)
422                        throw new IllegalArgumentException("Matrix.multiply: wrong row/col dimensions");
423                for (int i = 0; i < mA; i++) {
424                        for (int j = 0; j < nB; j++) {
425                                float s = 0;
426                                for (int k = 0; k < nA; k++) {
427                                        s = s + A[i][k] * B[k][j];
428                                }
429                                C[i][j] = s;
430                        }
431                }
432        }

656        public static float[] concatenate(float[]... xs) {
657                List<Float> vlist = new ArrayList<Float>();
658                for (float[] a : xs) {
659                        for (float val : a) {
660                                vlist.add(val);
661                        }
662                }
663                
664                float[] va = new float[vlist.size()];
665                int i = 0;
666                for (float val : vlist) {
667                        va[i] = val;
668                        i++;
669                }
670                return va;
671        }

692        public static double[] toHomogeneous(double[] cvec) {
693                double[] hvec = new double[cvec.length + 1];
694                for (int i = 0; i < cvec.length; i++) {

702                double[] cvec = new double[hvec.length - 1];
703                final double s = 1 / hvec[hvec.length - 1];     // TODO: check for zero factor
704                for (int i = 0; i < hvec.length - 1; i++) {
705                        cvec[i] = s * hvec[i];
706                }

720        public static float determinant3x3(final float[][] A) {
721                return
722                                A[0][0] * A[1][1] * A[2][2] +
723                                A[0][1] * A[1][2] * A[2][0] +
724                                A[0][2] * A[1][0] * A[2][1] -
725                                A[2][0] * A[1][1] * A[0][2] -
726                                A[2][1] * A[1][2] * A[0][0] -
727                                A[2][2] * A[1][0] * A[0][1] ;

730        public static double determinant3x3(final double[][] A) {
731                return 
732                        A[0][0] * A[1][1] * A[2][2] + 
733                        A[0][1] * A[1][2] * A[2][0]     + 
734                        A[0][2] * A[1][0] * A[2][1] - 
735                        A[2][0] * A[1][1] * A[0][2] - 
736                        A[2][1] * A[1][2] * A[0][0] - 
737                        A[2][2] * A[1][0] * A[0][1] ;
738        }

772        
773        public static double froebeniusNorm(final double[][] A) {
774                final int m = getNumberOfRows(A);
775                final int n = getNumberOfColumns(A);
776                double s = 0;
777                for (int i = 0; i < m; i++) {
778                        for (int j = 0; j < n; j++) {
779                                s = s + A[i][j] * A[i][j];
780                        }
781                }
782                return Math.sqrt(s);
783        }

787        public static double trace(final double[][] A) {
788                final int m = getNumberOfRows(A);
789                final int n = getNumberOfColumns(A);
790                if (m != n) 
791                        throw new IllegalArgumentException("square matrix expected");
792                double s = 0;
793                for (int i = 0; i < m; i++) {
794                                s = s + A[i][i];
795                }
796                return s;
797        }

938        public static double[] solve(final double[][] A, double[] b) {
939                RealMatrix AA = MatrixUtils.createRealMatrix(A);
940                RealVector bb = MatrixUtils.createRealVector(b);
941                DecompositionSolver solver = new LUDecomposition(AA).getSolver();
942                double[] x = null;
943                try {
944                        x = solver.solve(bb).toArray();
945                } catch (SingularMatrixException e) {}
946                return x;
947        }
