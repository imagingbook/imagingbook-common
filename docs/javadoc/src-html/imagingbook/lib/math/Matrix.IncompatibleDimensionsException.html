027/**
028 * <p>This class defines a set of static methods for calculations
029 * with vectors and matrices using native Java arrays without any enclosing 
030 * objects structures. 
031 * Matrices are simply two-dimensional arrays A[r][c], where r is the row index
032 * and c is the column index (as common in linear algebra). This means that
033 * matrices are really vectors of row vectors.
034 * Only arrays of type {@code float} and {@code double} are supported.
035 * All matrices are assumed to be rectangular (i.e., all rows are of equal length).</p>
036 * 
037 * <p>Methods named with a trailing 'D' (e.g., {@code multiplyD}) operate destructively,
038 * i.e., modify one of the passed arguments.</p>
039 * 
040 * <p>Most methods are self-explanatory and are therefore left undocumented.</p>
041 * 
042 * @author W. Burger
043 * @version 2019/01/05
044 */

328        /**
329         * Destructive version of {@link #multiply(double[], double[][])}.
330         * @param x a (row) vector of length m
331         * @param A matrix of size (m,n)
332         * @param y a (row) vector of length n
333         */
334        public static void multiplyD(final double[] x, final double[][] A, double[] y) {
335                if (x == y) 
336                        throw new SameSourceTargetException();
337                final int m = getNumberOfRows(A);
338                final int n = getNumberOfColumns(A);
339                if (x.length != m || y.length != n) 
340                        throw new IncompatibleDimensionsException();
341                for (int i = 0; i < n; i++) {
342                        double s = 0;
343                        for (int j = 0; j < m; j++) {
344                                s = s + x[j] * A[j][i];
345                        }
346                        y[i] = s;
347                }
348        }

433        // A * B -> C (destructive)
434        public static void multiplyD(final double[][] A, final double[][] B, final double[][] C) {
435                if (A == C || B == C) 
436                        throw new SameSourceTargetException();
437                final int mA = getNumberOfRows(A);
438                final int nA = getNumberOfColumns(A);
439                final int mB = getNumberOfRows(B);
440                final int nB = getNumberOfColumns(B);
441                if (nA != mB)
442                        throw new IncompatibleDimensionsException();
443                for (int i = 0; i < mA; i++) {
444                        for (int j = 0; j < nB; j++) {
445                                double s = 0;
446                                for (int k = 0; k < nA; k++) {
447                                        s = s + A[i][k] * B[k][j];
448                                }
449                                C[i][j] = s;
450                        }
451                }
452        }

487        /**
488         * Calculates and returns the dot (inner or scalar) product of two vectors,
489         * which must have the same length.
490         * @param x first vector
491         * @param y second vector
492         * @return the dot product
493         */
494        public static double dotProduct(final double[] x, final double[] y) {
495                if (!sameSize(x, y))
496                        throw new IncompatibleDimensionsException();
497                double sum = 0;
498                for (int i = 0; i < x.length; i++) {
499                        sum = sum + x[i] * y[i];
500                }
501                return sum;
502        }

504        // A is considered a column vector, B is a row vector, of length m, n, resp.
505        // returns a matrix M of size (m,n).
506        /**
507         * Calculates and returns the outer product of two vectors, which is a
508         * matrix of size (m,n), where m is the length of the first vector and
509         * m is the length of the second vector.
510         * @param x first vector (of length m)
511         * @param y second vector (of length n)
512         * @return the outer product (matrix)
513         */
514        public static double[][] outerProduct(final double[] x, final double[] y) {
515                final int m = x.length;
516                final int n = y.length;
517                final double[][] M = new double[m][n];
518                for (int i = 0; i < m; i++) {
519                        for (int j = 0; j < n; j++) {
520                                M[i][j] = x[i] * y[j];
521                        }
522                }
523                return M;
524        }

606         * @param A a matrix
607         * @return the norm of the matrix
608         */
609        public static double norm(final double[][] A) {
610                final int m = getNumberOfRows(A);
611                final int n = getNumberOfColumns(A);
612                double s = 0;
613                for (int i = 0; i < m; i++) {
614                        for (int j = 0; j < n; j++) {
615                                s = s + A[i][j] * A[i][j];
616                        }
617                }
618                return Math.sqrt(s);

902        /**
903         * Converts a Cartesian vector to an equivalent homogeneous
904         * vector, which contains an additional 1-element.
905         * @param xc a Cartesian vector
906         * @return an equivalent homogeneous vector
907         */
908        public static double[] toHomogeneous(double[] xc) {
909                double[] xh = new double[xc.length + 1];
910                for (int i = 0; i < xc.length; i++) {
911                        xh[i] = xc[i];
912                        xh[xh.length - 1] = 1;
913                }
914                return xh;
915        }
916        
917        /**
918         * Converts a homogeneous vector to its equivalent Cartesian
919         * vector, which is one element shorter.
920         * @param xh a homogeneous vector
921         * @return the equivalent Cartesian vector
922         */
923        public static double[] toCartesian(double[] xh) {
924                double[] xc = new double[xh.length - 1];
925                final double s = 1 / xh[xh.length - 1];
926                if (isZero(s))
927                        throw new DivideByZeroException();
928                for (int i = 0; i < xh.length - 1; i++) {
929                        xc[i] = s * xh[i];
930                }
931                return xc;
932        }

949                if (A.length != 3 || A[0].length != 3)
950                        throw new IncompatibleDimensionsException();
951                return
952                                A[0][0] * A[1][1] * A[2][2] +
953                                A[0][1] * A[1][2] * A[2][0] +
954                                A[0][2] * A[1][0] * A[2][1] -
955                                A[2][0] * A[1][1] * A[0][2] -
956                                A[2][1] * A[1][2] * A[0][0] -
957                                A[2][2] * A[1][0] * A[0][1] ;
